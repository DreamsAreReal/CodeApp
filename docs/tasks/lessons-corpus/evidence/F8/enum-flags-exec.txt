## c1 — [Flags]: combine with |, ToString, test with &

stdout: "Read, Write True False\n"   -> expect "Read, Write True False"

## c2 — underlying type is a CHOICE: enum : byte
$ POST run-csharp {enum Size : byte {S=1,M=2,L=4}; (int)Size.L, GetUnderlyingType.Name}

## c3 — default(E) = (E)0 even without a 0-member -> invalid value

stdout: "0 0 False\n"   -> expect "0 0 False"
